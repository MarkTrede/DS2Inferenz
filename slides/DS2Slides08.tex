\section{Stichproben und Statistiken}

\begin{frame}
\frametitle{Zufallsstichproben}
\framesubtitle{Kap.~4.1}
\textbf{Stichproben und Statistiken}\medskip
\begin{itemize}
\item Sei $X$ eine Zufallsvariable; wir interessieren uns für die
Verteilung von $X$, kennen aber die Verteilungsfunktion $F(x)$ nicht\pause
\item Um Informationen über die Verteilung zu gewinnen,\\
muss man $X$ beobachten\pause
\item Eine einzelne Beobachtung von $X$ reicht nicht aus,\\
um Informationen über die Verteilung zu gewinnen
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Zufallsstichproben}
\framesubtitle{Kap.~4.1}
\begin{block}{Einfache Stichprobe}
$X_{1},X_{2},\ldots ,X_{n}$ heißt einfache Stichprobe aus $X$ wenn
\begin{enumerate}
\item jedes $X_{i}$ wie $X$ verteilt ist und
\item $X_{1},X_{2},\ldots ,X_{n}$ stochastisch unabhängig sind
\end{enumerate}
\end{block}
Die Anzahl $n$ heißt Stichprobenumfang.
\end{frame}

\begin{frame}
\frametitle{Zufallsstichproben}
\framesubtitle{Kap.~4.1}
\begin{itemize}
\item \textbf{Achtung}: Die Definition stimmt nicht mit der 
umgangssprachlichen Verwendung des Wortes Stichprobe überein
\item Eine Stichprobe besteht \textbf{nicht} aus den tatsächlich
beobachteten Daten
\item Die tatsächlich beobachteten Daten seien $x_{1},\ldots ,x_{n}$\\ 
(Realisierung der Stichprobe, konkrete Stichprobe)
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Zufallsstichproben}
\framesubtitle{Kap.~4.1}
\begin{exampleblock}{Beispiel: Löhne}
\begin{itemize}
\item Sei $X$ der Lohn (Gehalt) einer zufällig ausgewählten
vollzeitbeschäftigten Person in Deutschland
\item Wir interessieren uns für $E(X)$ (also für den
Durchschnittslohn)
\item Es sollen $n=100$ Personen befragt werden
\end{itemize}
\end{exampleblock}
\end{frame}

\begin{frame}
\frametitle{Statistiken}
\framesubtitle{Kap.~4.2}
Sei $X_{1},\ldots ,X_{n}$ eine einfache Stichprobe aus $X$\\
und $g$ eine reellwertige Funktion mit $n$ Argumenten
\begin{block}{Statistik}
Die Zufallsvariable 
\[ T=g(X_1,\ldots ,X_n) \]
heißt Statistik oder Stichprobenfunktion
\end{block}
\end{frame}

\begin{frame}
\frametitle{Statistiken}
\framesubtitle{Kap.~4.2}
\begin{exampleblock}{Beispiel: Stichprobenmittel}
\[ \bar{X}=\frac{1}{n}\sum_{i=1}^{n}X_{i} \]
\end{exampleblock}
\begin{exampleblock}{Beispiel: Stichprobenvarianz}
\[ S^{2}=\frac{1}{n}\sum_{i=1}^{n}\left( X_{i}-\bar{X}\right) ^{2} \]
\end{exampleblock}
\end{frame}

\begin{frame}
\frametitle{Statistiken}
\framesubtitle{Kap.~4.3}
\textbf{Exkurs:} $\chi^2$- und $t$-Verteilung\medskip
\begin{itemize}
\item Seien $U_{1},\ldots ,U_{\nu }$ unabhängig $N(0,1)$-verteilt. 
Dann heißt die Zufallsvariable
\[ Q=\sum_{i=1}^{\nu }U_i^2 \]
$\chi^2$-verteilt mit $\nu$ Freiheitsgraden; $Q\sim\chi_\nu^2$
\item Plots der Dichtefunktion der $\chi^2$-Verteilung in R
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Statistiken}
\framesubtitle{Kap.~4.3}
\begin{itemize}
\item Seien $Q\sim\chi_\nu^2$ und $U\sim N(0,1)$ 
unabhängig. Dann heißt
\[ W=\frac{U}{\sqrt{Q/\nu }} \]
$t$-verteilt mit $\nu $ Freiheitsgraden; $W\sim t_\nu$
\item Plots der Dichtefunktion der $t$-Verteilung in R
\item Die Quantile der $t_{\nu }$ und $\chi _{\nu }^{2}$ sind tabelliert\\
und in R implementiert
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Statistiken bei normalverteilter Stichprobe}
\framesubtitle{Kap.~4.3}
\textbf{Verteilung des Stichprobenmittels}\medskip
\begin{itemize}
\item Wenn $X\sim N\left( \mu ,\sigma ^{2}\right) $ und $X_{1},\ldots ,X_{n}$
eine einfache Stichprobe aus $X$ ist, dann gilt
\begin{eqnarray*}
\bar{X} &\sim& N\left( \mu ,\frac{\sigma ^{2}}{n}\right) \\
\sqrt{n}\frac{\bar{X}-\mu }{\sigma } &\sim& N\left( 0,1\right)
\end{eqnarray*}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Statistiken bei normalverteilter Stichprobe}
\framesubtitle{Kap.~4.3}
\begin{itemize}
\item Ferner gilt
\[ \sqrt{n-1}\frac{\bar{X}-\mu }{S}\sim t_{n-1} \]
mit
\begin{eqnarray*}
S &=&\sqrt{S^2} \\
&=&\sqrt{\frac{1}{n}\sum_{i=1}^n(X_i-\bar{X})^2}
\end{eqnarray*}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Statistiken bei normalverteilter Stichprobe}
\framesubtitle{Kap.~4.3}
\textbf{Fazit}\medskip
\begin{itemize}
\item Die Verteilung des Stichprobenmittels $\bar{X}$ 
ist bekannt, \emph{wenn }man die Parameter $\mu $
und $\sigma ^{2}$ kennt
\item Problem: Die Parameter sind im Allgemeinen natürlich unbekannt
\item Wie kann man dieses Wissen trotzdem nutzen?
\end{itemize}
\end{frame}

\section{Schätzverfahren}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~4.3}
\begin{itemize}
\item Sei $X$ eine Zufallsvariable mit unbekannter\\
Verteilungsfunktion $F(x)$
\item Wir interessieren uns meist für einen (oder mehrere)\\
Parameter von $X$
\item Die Parameter werden mit Hilfe einer Stichprobe $X_{1},\ldots ,X_{n}$ geschätzt
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~4.3}
Wichtige interessierende Parameter sind z.B.:\medskip
\begin{itemize}
\item Erwartungswert $E(X)$
\item Varianz $V(X)$
\item Quantile $F_{X}^{-1}(p)$ für gegebenes $p\in (0,1)$
\item Werte der Verteilungsfunktion $F_{X}(x)$
\item Werte der Dichtefunktion $f_{X}(x)$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{block}{Schätzer (estimator)}
Sei $\theta $ ein unbekannter Parameter. Eine Statistik 
$\hat{\theta}(X_1,\ldots ,X_n)$ heißt Schätzer für $\theta$
\end{block}
\begin{block}{Schätzwert (estimate)}
Die Realisierung $\hat{\theta}(x_1,\ldots ,x_n)$ eines 
Schätzers heißt Schätzwert
\end{block}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{itemize}
\item Der Schätzer $\hat\theta=\hat\theta(X_1,\ldots,X_n)$ 
ist eine Zufallsvariable!
\item Der Schätzwert $\hat\theta=\hat\theta(x_1,\ldots,x_n)$ 
ist \textbf{keine} Zufallsvariable!\pause
\item \textbf{Achtung}: Die abkürzende Notation $\hat\theta$ bezeichnet
\begin{itemize}
\item entweder den Schätzer $\hat{\theta}(X_{1},\ldots ,X_{n})$
\item oder den Schätzwert $\hat{\theta}(x_{1},\ldots ,x_{n})$
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{itemize}
\item Schätzer für den Erwartungswert $\theta =E(X)$ 
aus der Stichprobe $X_{1},\ldots ,X_{n}$
\[ \hat{\theta}=\bar{X}=\frac{1}{n}\sum_{i=1}^{n}X_{i} \]
\item Schätzwert aus $x_{1},\ldots ,x_{n}$ (arithmetisches Mittel)
\[ \hat{\theta}=\bar{x}=\frac{1}{n}\sum_{i=1}^{n}x_{i} \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{itemize}
\item Schätzer für die Varianz $\theta =V(X)$ aus $X_{1},\ldots,X_{n}$
\[ \hat{\theta}=S^2=\frac{1}{n}\sum_{i=1}^n(X_i-\bar{X})^2 \]
\item Schätzwert aus $x_{1},\ldots ,x_{n}$ (empirische Varianz)
\[ \hat{\theta}=s^2=\frac{1}{n}\sum_{i=1}^{n}\left( x_{i}-\bar{x}\right) ^{2} \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
Damit eine Statistik $\hat{\theta}\left( X_{1},\ldots ,X_{n}\right)$
ein "`guter"' Schätzer für $\theta $ ist, sollte sie einige
Qualitätskriterien erfüllen\medskip
\begin{block}{Erwartungstreue (unbiasedness)}
Ein Schätzer $\hat{\theta}\left( X_{1},\ldots ,X_{n}\right) $ heißt
unverzerrt (unbiased) oder erwartungstreu für $\theta$, wenn
\[ E(\hat{\theta})=\theta \]
\end{block}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{itemize}
\item Gedankenexperiment: Wiederholte Stichprobe\pause
\item Ein erwartungstreuer Schätzer trifft nicht systematisch daneben\pause
\item Der Schätzer $\bar{X}$ ist unverzerrt für $E(X)$\pause
\item Spezialfall: $X$ Bernoulli-verteilt mit Parameter $\pi $;\\
der Schätzer $\hat{\pi}=\bar{X}=\frac{1}{n}\sum X_{i}$ ist erwartungstreu
\end{itemize}
\end{frame}

für $\sigma^2$:
\[ E(S^2) = \ldots \]
\item Korrigerte Stichprobenvarianz
\[ S^{\ast 2} = \hspace*{6cm} \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Schätzverfahren für Parameter}
\framesubtitle{Kap.~5.1}
\begin{itemize}
\item Der Schätzer $S^{\ast 2}$ ist erwartungstreu für $\sigma ^{2}$
\[ E\left( S^{\ast 2}\right) =\sigma ^{2} \]

\end{itemize}

\section{Intervallschätzung}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\textbf{Intervallschätzung}\medskip
\begin{itemize}
\item Punktschätzung: Schätzer liefert eine Zahl
\item Intervallschätzung: Schätzer liefert ein Intervall, in dem der
unbekannte Parameter mit großer Wahrscheinlichkeit liegt
\item Das Konfidenzniveau wird vorgegeben: $1-\alpha $
\item Vorteil: Die Genauigkeit der Schätzung wird deutlich
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
Sei $X$ eine Zufallsvariable, $\theta $ ein unbekannter Parameter 
und $X_1,\ldots ,X_n$ eine einfache Stichprobe\medskip
\begin{block}{Konfidenzintervall}
Ein Intervall $[\hat\theta_u(X_1,\ldots,X_n);\hat\theta_o(X_1,\ldots,X_n)]$ 
mit
\[ P(\hat\theta_u\leq \theta \leq \hat\theta_o) =1-\alpha \]
heißt Konfidenzintervall für $\theta $ zum Niveau $1-\alpha $
\end{block}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\begin{itemize}
\item Das Intervall $[\hat{\theta}_{u}\left( x_{1},\ldots ,x_{n}\right) ;%
\hat{\theta}_{o}\left( x_{1},\ldots ,x_{n}\right) ]$ heißt konkretes
Konfidenzintervall oder Wert des Konfidenzintervalls\pause
\item Die Grenzen $\hat{\theta}_{u}$ und $\hat{\theta}_{o}$ des
Konfidenzintervalls sind Zufallsvariablen; die Grenzen $\hat{\theta}_{u}$
und $\hat{\theta}_{o}$ des konkreten Konfidenzintervalls sind reelle Zahlen\pause
\item \textbf{Achtung:} Wahrscheinlichkeitsaussagen kann man nur über Konfidenzintervalle
machen, nicht über konkrete Konfidenzintervalle!
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\textbf{Konfindenzintervall für Erwartungswert $\mu$}\\\medskip
\begin{itemize}
\item \textbf{Fall 1}: $X$ normalverteilt, $\sigma^2$ bekannt
\[ \sqrt{n}\frac{\bar{X}-\mu }{\sigma}\sim N(0,1) ,\]
\item Also gilt für $c=u_{1-\alpha /2}$
\[ P\left( -c\leq \sqrt{n}\frac{\bar{X}-\mu }{\sigma }\leq c\right) =1-\alpha \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\begin{itemize}
\item Umformen ergibt
\[ P\left( \bar{X}-c\frac{\sigma }{\sqrt{n}}\leq \mu \leq \bar{X}
+c\frac{\sigma}{\sqrt{n}}\right) =1-\alpha \]
\item Das Konfidenzintervall zum Niveau $1-\alpha $ ist
\[ \bar{X} \pm u_{1-\frac{\alpha }{2}}\cdot \frac{\sigma}{\sqrt{n}} \]
\item Für $1-\alpha =0.95$ gilt $c=u_{0.975}=1.96$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}

Schokolade
\item $X$ sei normalverteilt mit bekannter Varianz $\sigma^2=4$ 
und unbekanntem Erwartungswert $\mu $
\item Der Erwartungswert $\mu $ soll geschätzt werden
\item 0.95-Konfidenzintervall für $\mu$ ?
\end{itemize}
\end{exampleblock}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\textbf{Konfidenzintervall für Erwartungswert $\mu$}\\\medskip
\begin{itemize}
\item \textbf{Fall 2}: $X$ \textbf{beliebig} verteilt, $n$ groß, $\sigma^2$ bekannt
\item Wegen des zentralen Grenzwertsatzes gilt für beliebig verteiltes 
$X$ bei großem $n$
\[ \sqrt{n}\frac{\bar{X}-\mu}{\sigma}\overset{appr}{\sim}N(0,1) \]
\end{itemize}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\begin{itemize}
\item Folglich ist
\[ \bar{X} \pm u_{1-\frac{\alpha}{2}}\cdot \frac{\sigma }{\sqrt{n}} \]
ein approximatives (asymptotisches) Konfidenzintervall für $\mu$ zum
Niveau $1-\alpha $, d.h.
\[ P\left( \bar{X}-u_{1-\frac{\alpha }{2}}\frac{\sigma }{\sqrt{n}}\leq \mu \leq 
\bar{X}+u_{1-\frac{\alpha }{2}}\frac{\sigma }{\sqrt{n}}\right) \approx 1-\alpha \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\textbf{Konfidenzintervall für Erwartungswert $\mu$}\\\medskip
\begin{itemize}
\item \textbf{Fall 3}: $X$ normalverteilt, $\sigma^2$ unbekannt
\[ \sqrt{n}\frac{\bar{X}-\mu }{S^{\ast }}\sim t_{n-1} \]
\item Also gilt für $c=t_{n-1,1-\alpha /2}$
\[ P\left( -c\leq \sqrt{n}\frac{\bar{X}-\mu }{S^{\ast }}\leq c\right) =1-\alpha  \]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\begin{itemize}
\item Umformen ergibt
\[ P\left( \bar{X}-c\frac{S^{\ast }}{\sqrt{n}}\leq \mu \leq \bar{X}+c\frac{%
S^{\ast }}{\sqrt{n}}\right) =1-\alpha \]
\item Konfidenzintervall für $\mu $ zum Niveau $1-\alpha$
\[ \bar{X}\pm t_{n-1,1-\frac{\alpha }{2}}\cdot \frac{S^{\ast }}{\sqrt{n}} \]
\end{itemize}
\end{frame}

\textbf{Wahl des Stichprobenumfangs $n$}\medskip
\begin{itemize}
\item Wie groß muss der Stichprobenumfang $n$ sein,\\
damit das Konfidenzintervall für $\mu$ eine bestimmte\\
Breite einhält?
\item Sei $2b$ die gewünschte maximale Breite
\item Sei $X$ normalverteilt und $\sigma^2$ bekannt
\item $(1-\alpha)$-Konfidenzintervall $\bar X\pm u_{1-\frac{\alpha}{2}}\cdot\frac{\sigma}{\sqrt{n}}$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Intervallschätzung}
\framesubtitle{Kap.~5.3}
\begin{itemize}
\item Aus
\[ b=u_{1-\frac{\alpha }{2}}\cdot \frac{\sigma }{\sqrt{n}} \]
folgt der Mindeststichprobenumfang
\[ n_{\min}=\frac{u_{1-\frac{\alpha }{2}}^{2}\cdot \sigma ^{2}}{b^{2}} \]
\item Gegebenenfalls muss $\sigma ^{2}$ vorher geschätzt werden
\end{itemize}
\end{frame}
